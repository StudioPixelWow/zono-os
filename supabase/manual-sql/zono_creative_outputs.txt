create table public.zono_creative_outputs (
  id                     uuid primary key default gen_random_uuid(),
  org_id                 uuid not null references public.organizations(id) on delete cascade,
  entity_type            text not null,
  entity_id              uuid not null,
  campaign_id            uuid references public.zono_campaigns(id) on delete set null,
  creative_asset_id      uuid references public.zono_creative_assets(id) on delete set null,
  copy_asset_id          uuid references public.zono_copy_assets(id) on delete set null,
  output_type            text not null,
  title                  text,
  status                 text not null default 'draft',
  preview_url            text,
  thumbnail_url          text,
  render_data            jsonb not null default '{}'::jsonb,
  generation_metadata    jsonb not null default '{}'::jsonb,
  brand_match_score      integer not null default 0 check (brand_match_score between 0 and 100),
  marketing_match_score  integer not null default 0 check (marketing_match_score between 0 and 100),
  readability_score      integer not null default 0 check (readability_score between 0 and 100),
  hierarchy_score        integer not null default 0 check (hierarchy_score between 0 and 100),
  conversion_score       integer not null default 0 check (conversion_score between 0 and 100),
  overall_score          integer not null default 0 check (overall_score between 0 and 100),
  is_approved            boolean not null default false,
  is_favorite            boolean not null default false,
  created_at             timestamptz not null default now(),
  updated_at             timestamptz not null default now()
);
create index zono_outputs_org_idx     on public.zono_creative_outputs(org_id);
create index zono_outputs_entity_idx  on public.zono_creative_outputs(entity_type, entity_id);
create index zono_outputs_asset_idx   on public.zono_creative_outputs(creative_asset_id);
create index zono_outputs_status_idx  on public.zono_creative_outputs(status);

create trigger trg_zono_outputs_updated before update on public.zono_creative_outputs for each row execute function public.set_updated_at();

alter table public.zono_creative_outputs enable row level security;
create policy "zono_creative_outputs_select" on public.zono_creative_outputs for select to authenticated using (org_id = public.current_org_id());
create policy "zono_creative_outputs_insert" on public.zono_creative_outputs for insert to authenticated with check (org_id = public.current_org_id() and public.has_min_role('agent'));
create policy "zono_creative_outputs_update" on public.zono_creative_outputs for update to authenticated using (org_id = public.current_org_id() and public.has_min_role('agent')) with check (org_id = public.current_org_id());
create policy "zono_creative_outputs_delete" on public.zono_creative_outputs for delete to authenticated using (org_id = public.current_org_id() and public.has_min_role('agent'));

grant select, insert, update, delete on public.zono_creative_outputs to authenticated;
grant all privileges on public.zono_creative_outputs to service_role;
