create table if not exists public.automation_copy_templates (
  id uuid primary key default gen_random_uuid(),
  template_key text not null unique,
  category text not null,
  voice text not null default 'agent',
  priority_label text,
  title_he text not null, subtitle_he text, short_description_he text, full_description_he text not null,
  task_title_he text, task_description_he text, agent_guidance_he text not null, manager_guidance_he text not null,
  revenue_impact_he text not null, urgency_reason_he text, expected_outcome_he text,
  decision_brain_summary_he text not null, success_definition_he text,
  client_draft_message_he text, portal_message_he text, website_message_he text, audit_log_text_he text,
  metadata jsonb not null default '{}'::jsonb,
  created_at timestamptz not null default now(), updated_at timestamptz not null default now()
);
create table if not exists public.automation_message_variants (
  id uuid primary key default gen_random_uuid(),
  template_key text not null, channel text not null, voice text not null default 'agent',
  message_he text not null, created_at timestamptz not null default now(),
  constraint amv_channel_chk check (channel in ('client','portal','website','sms_draft'))
);
create table if not exists public.automation_voices (
  voice_key text primary key, name_he text not null, description_he text not null, tone_he text not null, sort_order int not null default 0
);
create table if not exists public.automation_priority_labels (
  label_key text primary key, label_he text not null, description_he text not null, tone text not null default 'info', sort_order int not null default 0
);
create table if not exists public.automation_microcopy (
  id uuid primary key default gen_random_uuid(), scope text not null, copy_key text not null, text_he text not null,
  unique(scope, copy_key)
);
create index if not exists amv_key_idx on public.automation_message_variants(template_key);
create index if not exists act_cat_idx on public.automation_copy_templates(category);
