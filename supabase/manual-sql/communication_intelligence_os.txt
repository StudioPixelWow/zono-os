
create table public.communication_events (
  id                   uuid primary key default gen_random_uuid(),
  org_id               uuid not null references public.organizations(id) on delete cascade,
  thread_id            uuid references public.communication_threads(id) on delete set null,
  actor_user_id        uuid references public.users(id) on delete set null,
  source               text not null default 'manual',
  channel              text,
  direction            text,
  entity_type          text not null,
  entity_id            uuid not null,
  related_entity_type  text,
  related_entity_id    uuid,
  title                text,
  body                 text,
  transcript           text,
  is_voice_note        boolean not null default false,
  intent               text,
  sentiment            text,
  occurred_at          timestamptz not null default now(),
  metadata             jsonb not null default '{}'::jsonb,
  created_at           timestamptz not null default now()
);
create index comm_events_org_idx     on public.communication_events(org_id);
create index comm_events_entity_idx  on public.communication_events(entity_type, entity_id);
create index comm_events_thread_idx  on public.communication_events(thread_id);
create index comm_events_occurred_idx on public.communication_events(occurred_at);

create table public.communication_summaries (
  id                  uuid primary key default gen_random_uuid(),
  org_id              uuid not null references public.organizations(id) on delete cascade,
  entity_type         text not null,
  entity_id           uuid not null,
  thread_id           uuid references public.communication_threads(id) on delete set null,
  scope               text not null default 'relationship',
  summary_text        text,
  what_client_wants   text,
  what_changed        text,
  blocking_progress   text,
  next_step           text,
  key_points          jsonb not null default '[]'::jsonb,
  period_start        timestamptz,
  period_end          timestamptz,
  generated_at        timestamptz not null default now(),
  created_at          timestamptz not null default now()
);
create index comm_summaries_org_idx    on public.communication_summaries(org_id);
create index comm_summaries_entity_idx on public.communication_summaries(entity_type, entity_id);

create table public.communication_entities (
  id                  uuid primary key default gen_random_uuid(),
  org_id              uuid not null references public.organizations(id) on delete cascade,
  event_id            uuid references public.communication_events(id) on delete cascade,
  entity_type         text not null,
  entity_id           uuid not null,
  extracted_kind      text not null,
  raw_value           text,
  normalized_value    text,
  confidence_score    smallint not null default 60,
  created_at          timestamptz not null default now()
);
create index comm_entities_org_idx    on public.communication_entities(org_id);
create index comm_entities_entity_idx on public.communication_entities(entity_type, entity_id);
create index comm_entities_kind_idx   on public.communication_entities(extracted_kind);

create table public.communication_objections (

  event_id            uuid references public.communication_events(id) on delete set null,
  objection_type      text not null,
  severity            text not null default 'medium',

  resolved            boolean not null default false,
  resolution_method   text,
  detected_at         timestamptz not null default now(),
  resolved_at         timestamptz,
  created_at          timestamptz not null default now(),
  updated_at          timestamptz not null default now()
);
create index comm_objections_org_idx     on public.communication_objections(org_id);
create index comm_objections_entity_idx  on public.communication_objections(entity_type, entity_id);
create index comm_objections_type_idx    on public.communication_objections(objection_type);
create index comm_objections_resolved_idx on public.communication_objections(resolved);

create table public.communication_sentiment (

create table public.conversation_memory (
  id                  uuid primary key default gen_random_uuid(),
  org_id              uuid not null references public.organizations(id) on delete cascade,
  thread_id           uuid references public.communication_threads(id) on delete cascade,
  entity_type         text not null,
  entity_id           uuid not null,
  last_summary        text,
  last_intent         text,
  last_sentiment      text,
  open_loops          jsonb not null default '[]'::jsonb,
  established_facts    jsonb not null default '[]'::jsonb,

  last_event_at       timestamptz,
  created_at          timestamptz not null default now(),
  updated_at          timestamptz not null default now(),

);
create index conv_memory_org_idx    on public.conversation_memory(org_id);
create index conv_memory_entity_idx on public.conversation_memory(entity_type, entity_id);

create trigger trg_comm_objections_updated before update on public.communication_objections
  for each row execute function public.set_updated_at();
create trigger trg_comm_risks_updated before update on public.communication_risks
  for each row execute function public.set_updated_at();
create trigger trg_comm_opps_updated before update on public.communication_opportunities
  for each row execute function public.set_updated_at();
create trigger trg_client_memory_updated before update on public.client_memory
  for each row execute function public.set_updated_at();
create trigger trg_conv_memory_updated before update on public.conversation_memory
  for each row execute function public.set_updated_at();

do $$
declare t text;
  tbls text[] := array[
    'communication_events','communication_summaries','communication_entities',
    'communication_objections','communication_sentiment','communication_intents',
    'communication_risks','communication_opportunities','client_memory','conversation_memory'
  ];
begin
  foreach t in array tbls loop
    execute format('alter table public.%I enable row level security;', t);
    execute format(
      'create policy "%1$s_select" on public.%1$I for select to authenticated '
      || 'using (org_id = public.current_org_id());', t);
    execute format(
      'create policy "%1$s_insert" on public.%1$I for insert to authenticated '
      || 'with check (org_id = public.current_org_id() and public.has_min_role(''agent''));', t);
    execute format(
      'create policy "%1$s_update" on public.%1$I for update to authenticated '
      || 'using (org_id = public.current_org_id() and public.has_min_role(''agent'')) '
      || 'with check (org_id = public.current_org_id());', t);
    execute format(
      'create policy "%1$s_delete" on public.%1$I for delete to authenticated '
      || 'using (org_id = public.current_org_id() and public.has_min_role(''manager''));', t);
  end loop;
end $$;

grant select, insert, update, delete on
  public.communication_events, public.communication_summaries, public.communication_entities,
  public.communication_objections, public.communication_sentiment, public.communication_intents,
  public.communication_risks, public.communication_opportunities, public.client_memory,
  public.conversation_memory
  to authenticated;
grant all privileges on
  public.communication_events, public.communication_summaries, public.communication_entities,
  public.communication_objections, public.communication_sentiment, public.communication_intents,
  public.communication_risks, public.communication_opportunities, public.client_memory,
  public.conversation_memory
  to service_role;
